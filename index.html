trait HasLength {
  fn length(&self) -> f64;
}
					

impl HasLength for Vec3 {
  fn length(&self) -> f64 {
    let x = self.x;
    let y = self.y;
    let z = self.z;
    (x * x + y * y + z * z).sqrt()
  }
}
