trait Mult < RHS = Self> {
  type Output;
  pub fn mul(self, rhs: RHS) -> Self::Output;
}
			

impl Mult< f64> for Vec3 {
  type Output = Vec3;

  fn mul(self, val: f64) -> Vec3 {
    Vec3::new(self.x * val,
  	      self.y * val,
  	      self.z * val)
  }
}
